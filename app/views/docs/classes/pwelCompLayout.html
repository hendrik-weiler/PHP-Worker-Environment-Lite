Description

Making possible set a layout.

Example: 

//Create a instance with layoutfile given as parameter
new PWEL_COMPONENT_LAYOUT("layoutfile.php");

Functions
  • addVariables Adds variables by displaying the layout
  • disableLayout Disables the layout
  • enableLayout Enables the layout
addVariables()
Description:
* Adds all variables which will be used in the layout
* @var object/array $variables
class start extends PWEL_CONTROLLER {
    function index() {
        //Creates variables
        $this->varNumberOne = "im a test";
        $this->varNumberTwo = "im another test";
    }
    
    public function __destruct() {
        //They will be added on the displaying process
        PWEL_COMPONENT_LAYOUT::addVariables($this);
    }
}

disableLayout()
Description:
* Disables the function on controller method
class start extends PWEL_CONTROLLER {
    function index() {
        //Disable the layout
        PWEL_COMPONENT_LAYOUT::disableLayout();
        
        //Display other stuff
        $this->display("otherstuff");
    }
}

disableLayout()
Description:
* Enables the function on controller method
class start extends PWEL_CONTROLLER {
    function index() {
        //Disable the layout
        PWEL_COMPONENT_LAYOUT::disableLayout();
        
        //Display other stuff
        $this->display("otherstuff");
        
        //Enable the layout
        //This isnt a good situation but this method exist
        PWEL_COMPONENT_LAYOUT::enableLayout();        
    }
}
